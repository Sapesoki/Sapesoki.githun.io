    <label for="tipoInteres">Selecciona el tipo de interés:</label>
    <select id="tipoInteres" onchange="calcularInteres()">
        <option value="simple">Interés Simple</option>
        <option value="compuesto">Interés Compuesto</option>
    </select>

    <!-- Campos de entrada -->
    <label for="capital">Capital Inicial ($):</label>
    <input type="number" id="capital" value="1000" min="0" onchange="calcularInteres()">

    <label for="tasa">Tasa de Interés Anual (%):</label>
    <input type="number" id="tasa" value="5" min="0" onchange="calcularInteres()">

    <label for="anos">Tiempo en años:</label>
    <input type="number" id="anos" value="3" min="0" onchange="calcularInteres()">

    <!-- Resultados -->
    <div class="result">
        <p><strong>Monto Principal: $<span id="montoPrincipal">1000.00</span></strong></p>
        <p><strong>Interés Total: $<span id="interesTotal">0.00</span></strong></p>
        <p><strong>Valor Total: $<span id="valorTotal">1000.00</span></strong></p>
    </div>

<script>
    function calcularInteres() {
        var tipoInteres = document.getElementById("tipoInteres").value;
        var capital = parseFloat(document.getElementById("capital").value);
        var tasa = parseFloat(document.getElementById("tasa").value);
        var anos = parseFloat(document.getElementById("anos").value);
        var interesTotal, valorTotal;

        if (tipoInteres === "simple") {
            // Cálculo de interés simple
            interesTotal = (capital * tasa * anos) / 100;
            valorTotal = capital + interesTotal;
        } else {
            // Cálculo de interés compuesto
            valorTotal = capital * Math.pow((1 + tasa / 100), anos);
            interesTotal = valorTotal - capital;
        }

        // Mostrar resultados
        document.getElementById("montoPrincipal").innerText = capital.toFixed(2);
        document.getElementById("interesTotal").innerText = interesTotal.toFixed(2);
        document.getElementById("valorTotal").innerText = valorTotal.toFixed(2);
    }

    // Calcular el interés por primera vez al cargar la página
    calcularInteres();
</script>
